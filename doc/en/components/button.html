Button ​

A common component used to trigger actions.

Basic Usage ​

html
<!-- Basic Usage -->
<button class="t-btn">Default Button</button>
<button class="t-btn t-btn-primary">Primary Button</button>
<button class="t-btn t-btn-success">Success Button</button>
<button class="t-btn t-btn-warning">Warning Button</button>
<button class="t-btn t-btn-danger">Danger Button</button>

Plain Buttons ​

html
<!-- Plain Buttons -->
<button class="t-btn t-btn-plain">Plain Button</button>
<button class="t-btn t-btn-plain t-btn-primary">Primary Button</button>
<button class="t-btn t-btn-plain t-btn-success">Success Button</button>
<button class="t-btn t-btn-plain t-btn-warning">Warning Button</button>
<button class="t-btn t-btn-plain t-btn-danger">Danger Button</button>

Round Buttons ​

html
<!-- Round Buttons -->
<button class="t-btn t-btn-round">Round Button</button>
<button class="t-btn t-btn-round t-btn-primary">Primary Button</button>
<button class="t-btn t-btn-round t-btn-success">Success Button</button>

Circle Buttons ​

html
<!-- Circle Buttons -->
